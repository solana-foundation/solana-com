---
title: How to Get Test SOL Through Airdrops
description: Learn how to get test SOL for development purposes.
---

When building on Solana, you need SOL to pay for transaction fees and create
accounts to store data.

For development, you can get free test SOL through airdrops on non-mainnet
environments like devnet, testnet, or your local validator.

<CodeTabs>

```ts !! title="Kit"
// !collapse(1:9) collapsed

import {
  airdropFactory,
  createSolanaRpc,
  createSolanaRpcSubscriptions,
  lamports,
  generateKeyPairSigner,
} from "@solana/web3.js";

const rpc = createSolanaRpc("http://127.0.0.1:8899");
const rpcSubscriptions = createSolanaRpcSubscriptions("ws://127.0.0.1:8900");

const wallet = await generateKeyPairSigner();
const transactionSignature = await airdropFactory({ rpc, rpcSubscriptions })({
  recipientAddress: wallet.address,
  lamports: lamports(5_000_000_000n),
  commitment: "confirmed",
});
// !collapse(1:7) collapsed

const balance = await rpc.getBalance(wallet.address).send();
console.log("Account SOL Balance:", balance.value / 1_000_000_000n);
console.log(
  "Transaction Signature:",
  `https://explorer.solana.com/tx/${transactionSignature}?cluster=custom`,
);
```

```ts !! title="Legacy"
// !collapse(1:3) collapsed

import { Connection, LAMPORTS_PER_SOL, Keypair } from "@solana/web3.js";

const connection = new Connection("http://127.0.0.1:8899", "confirmed");

const wallet = Keypair.generate();
const transactionSignature = await connection.requestAirdrop(
  wallet.publicKey,
  5 * LAMPORTS_PER_SOL,
);
// !collapse(1:20) collapsed

const { blockhash, lastValidBlockHeight } =
  await connection.getLatestBlockhash();

await connection.confirmTransaction(
  {
    blockhash,
    lastValidBlockHeight,
    signature: transactionSignature,
  },
  "confirmed",
);

const balance = await connection.getBalance(wallet.publicKey);
console.log("Account SOL Balance:", balance / LAMPORTS_PER_SOL);

console.log(
  "Transaction Signature:",
  `https://explorer.solana.com/tx/${transactionSignature}?cluster=custom`,
);
```

```rs !! title="Rust"
// !collapse(1:9) collapsed

use anyhow::Result;
use solana_client::rpc_client::RpcClient;
use solana_sdk::{
    commitment_config::CommitmentConfig, native_token::LAMPORTS_PER_SOL, signature::Keypair,
    signer::Signer,
};

fn main() -> Result<()> {
    let client = RpcClient::new_with_commitment(
        String::from("http://127.0.0.1:8899"),
        CommitmentConfig::confirmed(),
    );

    let wallet = Keypair::new();
    let transaction_signature = client.request_airdrop(&wallet.pubkey(), 5 * LAMPORTS_PER_SOL)?;
    // !collapse(1:15) collapsed

    loop {
        if client.confirm_transaction(&transaction_signature)? {
            break;
        }
    }

    let balance = client.get_balance(&wallet.pubkey())?;
    println!("Account Balance: {}", balance / LAMPORTS_PER_SOL);
    println!(
        "Transaction Signature: https://explorer.solana.com/tx/{}?cluster=custom",
        transaction_signature
    );
    Ok(())
}
```
