h1: getMaxRetransmitSlot RPC Method

Returns the highest slot this node has observed in retransmit.

<Callout type="info" title="Source">
  [`get_max_retransmit_slot`](https://github.com/anza-xyz/agave/blob/v3.1.8/rpc/src/rpc.rs#L960)
</Callout>

```jsonc !!request curl

```

```ts !!request title="Kit"
import { createSolanaRpc } from "@solana/kit";

const rpc_url = "https://api.devnet.solana.com";
const rpc = createSolanaRpc(rpc_url);

let maxRetransmitSlot = await rpc.getMaxRetransmitSlot().send();

console.log(maxRetransmitSlot);
```

```rs !!request title="Rust"
use anyhow::Result;
use solana_client::nonblocking::rpc_client::RpcClient;
use solana_commitment_config::CommitmentConfig;

#[tokio::main]
async fn main() -> Result<()> {
    let client = RpcClient::new_with_commitment(
        String::from("https://api.devnet.solana.com"),
        CommitmentConfig::confirmed(),
    );

    let max_retransmit_slot = client.get_max_retransmit_slot().await?;

    println!("{:#?}", max_retransmit_slot);

    Ok(())
}
```
